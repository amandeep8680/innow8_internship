# 1. read_only=True
This makes a field only appear in responses, not accepted in requests.
class UserSerializer(serializers.ModelSerializer):
    id = serializers.IntegerField(read_only=True)

    class Meta:
        model = User
        fields = ['id', 'username', 'password']

✔ Client cannot send id
✔ API will return id

# 🔹 2. write_only=True
This makes a field only accepted in input, but never shown in output.

class UserSerializer(serializers.ModelSerializer):
    password = serializers.CharField(write_only=True)
    class Meta:
        model = User
        fields = ['username', 'password']
✔ Client must send password
❌ API won’t return password


# 🔹 3. Using extra_kwargs (cleaner way)

Instead of redefining fields, you can do:

class UserSerializer(serializers.ModelSerializer):
    class Meta:
        model = User
        fields = ['id', 'username', 'password']
        extra_kwargs = {
            'id': {'read_only': True},
            'password': {'write_only': True}
        }



# 🔹 4. Read-only fields list
Shortcut for multiple read-only fields:
class Meta:
    model = User
    fields = ['id', 'username']
    read_only_fields = ['id']


    
# 🔹 5. Dynamic behavior (advanced)

You can even change behavior based on request:

def get_fields(self):
    fields = super().get_fields()
    if self.context['request'].method == 'GET':
        fields['email'].read_only = True
    return fields



#  ⚠️ Important Notes
read_only fields are ignored during validation
write_only fields are not included in serialized output
Useful for:
Passwords 🔒
Auto-generated fields (id, timestamps)
Sensitive data
